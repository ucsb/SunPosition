Calculates and displays current subsolar point, based on time and time zone of the calling location
function SunPositionNow()
cd = datetime('now','TimeZone','local');
[declin,radius_vector,solar_lon,eqTime] = EarthEphemeris(cd);
fprintf('DateTime %s\nTimeZone %s\n',datestr(cd),cd.TimeZone)
fprintf('declination (deg) %f\nsolar longitude (deg) %f\nradius vector (AU) %f\nequation of time (min) %f\n',...
declin,solar_lon,radius_vector,eqTime)
end
