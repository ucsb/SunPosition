show image of daylight hours as a function of latitude and date
function GlobalDaylight()
latitude = [90:-0.5:-90];
dayno = 1:365;
x = datenum('12/31/2019');
[ll,mm] = ndgrid(latitude,x+dayno);
[r,s] = sunRiseSet(ll,0,mm);
daylight = 24*(datenum(s)-datenum(r));
daylight(isnan(daylight)) = 0;
imagesc(x+dayno,latitude,daylight)
set(gca,'YDir','normal')
datetick('x','mmm');
ylabel('latitude')
title('daylight hours')
colorbar
end
